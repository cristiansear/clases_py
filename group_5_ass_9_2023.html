    alterna.index.names = ['']
    locals()[f'Dengue_{year}'] = alterna

Dengue_semanal = Dengue_peru.loc[:,["Año", "Semana", "Casos", "Ubigeo"]]

Dengue_semanal["Trimestre"] = pd.Series(dtype=int)

for index,row in Dengue_semanal.iterrows():
    if Dengue_semanal.loc[index,"Semana"] <= 13:
        Dengue_semanal.loc[index,"Trimestre"] = 1
    elif Dengue_semanal.loc[index,"Semana"] <=26:
        Dengue_semanal.loc[index,"Trimestre"] = 2
    elif Dengue_semanal.loc[index,"Semana"] <=39:
        Dengue_semanal.loc[index,"Trimestre"] = 3
