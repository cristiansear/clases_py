def gen_cohot( series, size = 5 ):
    
    min1 = int( ( ( series.min() ) / size) ) * size
    max1 = int( np.ceil( ( series.max() ) / size) ) * size
    result = np.arange( min1, max1 + 1 , 5 )
    labels = np.ceil( ( result[ 1: ] + result[ :-1 ] ) / 2 ).astype( int )
    
    return result, labels
